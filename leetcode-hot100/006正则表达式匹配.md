## 题目描述

给你一个字符串 s 和一个字符规律 p，请你来实现一个支持 '.' 和 '*' 的正则表达式匹配。

'.' 匹配任意单个字符
'*' 匹配零个或多个前面的那一个元素
所谓匹配，是要涵盖 整个 字符串 s的，而不是部分字符串。

class Solution {
public:
    bool isMatch_one(char a, char b){
        if(a==NULL || b==NULL) return false;

        if(b == '.') return true;
        else if(a == b) return true;
        else return false;
    }
    void push_stack(stack<char> &st, string s){
        for(auto it = s.begin(); it!=s.end(); ++it){
            st.push(*it);
        }
    }
    void func(stack<char> s1, stack<char> s2){
         while(!s1.empty() && isMatch_one(s1.top(), s2.top())){
            s1.pop();
        }
    }
    bool isMatch(string s, string p) {
        stack<char> s1;
        stack<char> s2;
        push_stack(s1, s);
        push_stack(s2, p);

        while(!s1.empty() && !s2.empty()){
            if(s2.top() == '*'){
                s2.pop();
                func(s1, s2);
                s2.pop(); //当前s2匹配成功，出栈
            }else {
                if(isMatch_one(s1.top(), s2.top())){
                    s1.pop();
                    s2.pop();
                }else{
                    return false;
                }
            }
        }
        if(!s1.empty()) return false;
        else{
            while(!s2.empty()){ 
                 if(s2.top()!='*') return false;
                 else {
                    s2.pop();
                    s2.pop();
                }
            }
            return true;
           
        }
    }
};
